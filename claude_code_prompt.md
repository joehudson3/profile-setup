```yaml
  generate:

      - uses: Platane/snk@v3
        with:
          github_user_name: joehudson3
          outputs: |
            dist/snake.svg
            dist/snake-dark.svg?palette=github-dark

        with:
          target_branch: output
          build_dir: dist
        env:
          GITHUB_TOKEN: ${{ secrets.GITHUB_TOKEN }}
```
